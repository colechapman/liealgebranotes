\section*{Universal Enveloping Algebra}

\textbf{Definition 1.}
The \emph{universal enveloping algebra} of a Lie algebra $L$ is a pair
$U(L), \iota$ where:

\begin{itemize}
\item $U(L)$ is an associative algebra with unit,
\item $\iota : L \to U(L)^-$ is a Lie algebra homomorphism,
\end{itemize}

satisfying the following universal property:

For any associative algebra $A$ and any Lie algebra homomorphism
$\varphi : L \to A^-$, there exists a unique associative algebra
homomorphism $\tilde{\varphi} : U(L) \to A$ such that
\[
\varphi = \tilde{\varphi} \circ \iota.
\]

In particular, every representation of $L$ naturally extends to a
representation of $U(L)$, and every homomorphism of $L$-representations
is also a homomorphism of $U(L)$-representations.

\textbf{Proposition 1.}
$U(L)$ is unique up to isomorphism.

\textbf{Proposition 2.}
$U(L)$ exists for every Lie algebra $L$.

\textbf{Construction.}

Let
\[
T(L) = \mathbb{C} \oplus L \oplus L \otimes L \oplus \cdots
\]
be the tensor algebra of $L$.

Let $J$ be the two-sided ideal generated by elements
\[
x \otimes y - y \otimes x - [x,y], \quad x,y \in L.
\]

Then
\[
U(L) := T(L)/J.
\]

If $x_1,\dots,x_n$ is a basis of $L$ and
\[
[x_i,x_j] = \sum_k c_{ij}^k x_k,
\]
then $U(L)$ is the associative algebra with generators
$x_1,\dots,x_n$ and relations
\[
x_i x_j - x_j x_i = \sum_k c_{ij}^k x_k.
\]

\textbf{Example.}
If $L$ is abelian with basis $x_1,\dots,x_n$, then
\[
U(L) = S(L) = \mathbb{C}[x_1,\dots,x_n].
\]

\section*{Comultiplication}

\textbf{Definition 2.}
A \emph{comultiplication} on an associative algebra $A$ is an algebra
homomorphism
\[
\Delta : A \to A \otimes A
\]
such that
\[
(\Delta \otimes 1)\circ \Delta = (1 \otimes \Delta)\circ \Delta.
\]

An associative algebra with comultiplication is called a \emph{bialgebra}.

\textbf{Proposition 3.}
The map
\[
\Delta(\iota(x)) = \iota(x) \otimes 1 + 1 \otimes \iota(x)
\]
extends to a (cocommutative) comultiplication on $U(L)$.

\textbf{Example.}
The group algebra $\mathbb{C}G$ has comultiplication
\[
\Delta(g) = g \otimes g.
\]

\section*{Graded and Filtered Algebras}

\textbf{Definition 3.}
A (nonnegative $\mathbb{Z}$-)grading of $A$ is a decomposition
\[
A = \bigoplus_{n\ge0} A_n
\]
such that
\[
A_n A_m \subset A_{n+m}.
\]

Examples:

\begin{itemize}
\item Tensor algebra: $T(V)_n = V^{\otimes n}$.
\item Symmetric algebra: $S(V)_n = S^n V$.
\end{itemize}

\textbf{Definition 4.}
A filtration is an increasing chain
\[
A = \bigcup_{n\ge0} A^{(n)},
\quad
A^{(n-1)} \subset A^{(n)},
\]
such that
\[
A^{(n)} A^{(m)} \subset A^{(n+m)}.
\]

For $U(L)$, the filtration is by degree in generators.

\textbf{Definition 5.}
The associated graded algebra is
\[
\mathrm{gr} A = \bigoplus_{n\ge0} A^{(n)}/A^{(n-1)}.
\]

\textbf{Proposition 4.}
$\mathrm{gr}\,U(L)$ is a commutative algebra generated by $L$.

Thus there is a homomorphism
\[
S(L) \to \mathrm{gr}\,U(L).
\]

\section*{Poincaré–Birkhoff–Witt Theorem}

\textbf{Theorem (PBW).}
\[
\mathrm{gr}\,U(L) = S(L).
\]

Equivalently, if $x_1,\dots,x_n$ is a basis of $L$, then
monomials
\[
x_1^{k_1}\cdots x_n^{k_n}
\]
form a basis of $U(L)$.

\subsection*{Proof Sketch}

\textbf{Lemma 1.}
If the theorem holds for $L$ and $I$ is an ideal, then it holds for $L/I$.

\textbf{Lemma 2.}
If $\iota : L \to U(L)$ is injective, then PBW holds for $L$.

\textbf{Lemma 3.}
For the free Lie algebra $L(x_1,\dots,x_n)$,
\[
U(L) = \mathbb{C}\langle x_1,\dots,x_n\rangle
\]
(the free associative algebra).

\textbf{Lemma 4.}
For the free Lie algebra, $\iota$ is injective.

\textbf{Conclusion.}
Every Lie algebra is a quotient of a free Lie algebra.
Hence PBW holds for all Lie algebras.
